<script>
  // Collect all form inputs into a JSON object
  function collectFormInputs() {
    const inputs = {};
    const formElements = document.querySelectorAll('form input[type="number"], form input[type="text"], form select');

    formElements.forEach(input => {
      inputs[input.name] = input.value;
    });

    return inputs;
  }

  // Function to call your Flask API and save the scenario
  function saveScenario(scenarioName, inputsJson) {
    fetch('/scenarios/save', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ scenario_name: scenarioName, inputs_json: inputsJson })
    })
    .then(response => response.json())
    .then(data => {
      if (data.message) {
        alert('Scenario saved successfully!');
      } else if (data.error) {
        alert('Error: ' + data.error);
      }
    })
    .catch(error => alert('Network error: ' + error));
  }

  // Trigger Save Scenario on button click
  function triggerSaveScenario() {
    const scenarioName = document.getElementById('scenario_name_input').value.trim();

    if (!scenarioName) {
      alert('Please enter a scenario name before saving.');
      return;
    }

    const inputsJson = collectFormInputs();
    saveScenario(scenarioName, inputsJson);
  }
</script>


<script>
  // Fetch saved scenarios on page load
  document.addEventListener('DOMContentLoaded', () => {
    fetch('/scenarios/list')
      .then(response => response.json())
      .then(data => {
        const select = document.getElementById('load_scenario_select');
        data.forEach(scenario => {
          const option = document.createElement('option');
          option.value = scenario.id;
          option.textContent = scenario.scenario_name;
          select.appendChild(option);
        });
      })
      .catch(err => console.error('Error loading scenarios:', err));
  });

  function loadSelectedScenario() {
    const select = document.getElementById('load_scenario_select');
    const scenarioId = select.value;
    if (!scenarioId) {
      alert('Please select a scenario to load.');
      return;
    }

    fetch(`/scenarios/load/${scenarioId}`)
      .then(response => {
        if (!response.ok) throw new Error('Scenario not found');
        return response.json();
      })
      .then(data => {
        const inputs = data.inputs_json;

        // Iterate all keys in inputs JSON and set form values
        Object.entries(inputs).forEach(([key, value]) => {
          const el = document.querySelector(`[name="${key}"]`);
          if (el) {
            if (el.tagName.toLowerCase() === 'select') {
              el.value = value.toString();
            } else {
              el.value = value;
            }
          }
        });

        alert(`Loaded scenario: ${data.scenario_name}`);
      })
      .catch(err => alert('Failed to load scenario: ' + err.message));
  }
</script>
